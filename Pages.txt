##BlogTitles.jsx
import { useAuth } from '@clerk/clerk-react'
import { Hash, Sparkles } from 'lucide-react'
import React, { useState } from 'react'
import toast from 'react-hot-toast'
import Markdown from 'react-markdown'
import axios from 'axios'

axios.defaults.baseURL = import.meta.env.VITE_BASE_URL

const BlogTitles = () => {
  const blogCategories = [
    'General',
    'Technology',
    'Business',
    'Health',
    'Lifestyle',
    'Education',
    'Travel',
    'Food',
  ]

  const [selectedCategory, setSelectedCategory] = useState('General')
  const [input, setInput] = useState('')
  const [loading, setLoading] = useState(false)
  const [content, setContent] = useState('')

  const { getToken } = useAuth()

  const formatAITitles = (text) => {
    if (!text) return ''
    
    const lines = text.split('\n').filter(line => line.trim())
    const formattedLines = []
    
    lines.forEach(line => {
      const trimmedLine = line.trim()
      
      // Check if line is likely a section header
      const isHeader = 
        trimmedLine.endsWith(':') ||
        trimmedLine.toLowerCase().startsWith('for ') ||
        trimmedLine.toLowerCase().includes('section') ||
        trimmedLine.toLowerCase().includes('category') ||
        (trimmedLine.length < 60 && !trimmedLine.includes('-') && !trimmedLine.includes('*'))
      
      // Check if line is likely a title (longer text)
      const isTitle = 
        trimmedLine.length > 30 ||
        trimmedLine.includes('•') ||
        trimmedLine.includes('- ') ||
        trimmedLine.includes('* ') ||
        (trimmedLine.length > 15 && !trimmedLine.includes(':') && !isHeader)
      
      // Handle numbered items
      const isNumbered = /^\d+\.\s/.test(trimmedLine)
      
      if (isHeader) {
        // Clean up header text
        const cleanHeader = trimmedLine
          .replace(/:/g, '')
          .replace(/^(for|section|categories?)\s*/i, '')
          .trim()
        if (cleanHeader) {
          formattedLines.push(`## ${cleanHeader}`)
        }
      } else if (isNumbered) {
        // Convert numbered items to bullet points
        const cleanTitle = trimmedLine.replace(/^\d+\.\s/, '')
        formattedLines.push(`- ${cleanTitle}`)
      } else if (isTitle) {
        // Ensure it's a bullet point
        const cleanTitle = trimmedLine.replace(/^[•\-*]\s*/, '')
        formattedLines.push(`- ${cleanTitle}`)
      } else if (trimmedLine) {
        // Default fallback - keep as is
        formattedLines.push(trimmedLine)
      }
    })
    
    // If formatting didn't produce clear sections, fall back to a simpler approach
    if (formattedLines.length === 0 || formattedLines.every(line => line.startsWith('-'))) {
      return lines
        .map(line => {
          const trimmed = line.trim()
          if (trimmed.length > 40 && !trimmed.startsWith('-')) {
            return `- ${trimmed}`
          }
          return trimmed
        })
        .filter(line => line)
        .join('\n\n')
    }
    
    return formattedLines.join('\n\n')
  }

  const onSubmitHandler = async (e) => {
    e.preventDefault()
    try {
      setLoading(true)
      const prompt = `Generate blog titles for "${input}" in the "${selectedCategory}" category.
      Organize the output into clear sections with headings.
      Each section should have 3-5 blog titles.
      Use proper formatting with sections and bullet points.`

      const { data } = await axios.post(
        '/api/ai/generate-blog-title',
        { prompt },
        { headers: { Authorization: `Bearer ${await getToken()}` } }
      )

      if (data.success) {
        const formattedContent = formatAITitles(data.content)
        setContent(formattedContent)
      } else {
        toast.error(data.message)
      }
    } catch (error) {
      toast.error(error.message)
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="h-full overflow-y-auto p-8 flex items-start flex-wrap gap-8 text-slate-700 bg-gradient-to-br from-pink-50 via-white to-purple-50 transition-all duration-300">
      <form
        onSubmit={onSubmitHandler}
        className="w-full max-w-lg p-8 bg-white rounded-xl border border-gray-200 shadow-sm hover:shadow-md transition-all duration-300 ease-out"
      >
        <div className="flex items-center gap-3 mb-4">
          <Sparkles className="w-6 h-6 text-[#D946EF] animate-pulse" />
          <h1 className="text-2xl font-semibold text-gray-800">
            AI Blog Generator
          </h1>
        </div>

        <label className="block mt-6 text-sm font-medium text-gray-700">
          Keyword
        </label>
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          className="w-full p-3 mt-2 text-sm rounded-lg border border-gray-300 outline-none placeholder:text-gray-400 focus:ring-2 focus:ring-[#D946EF] focus:border-[#D946EF] transition-all duration-200 ease-in-out"
          placeholder="The Future of Artificial Intelligence..."
          required
        />

        <p className="mt-6 text-sm font-medium text-gray-700">Category</p>
        <div className="mt-3 flex gap-3 flex-wrap">
          {blogCategories.map((item) => {
            const isSelected = selectedCategory === item
            return (
              <span
                key={item}
                onClick={() => setSelectedCategory(item)}
                className={`relative text-sm px-5 py-2 rounded-full border cursor-pointer select-none font-medium transition-all duration-300 ease-out transform ${
                  isSelected
                    ? 'bg-gradient-to-r from-[#EC4899]/20 to-[#8B5CF6]/20 text-[#8B5CF6] border-[#E9D5FF] shadow-inner scale-[1.03]'
                    : 'bg-white/60 text-gray-700 border-gray-300 hover:bg-gradient-to-r hover:from-[#EC4899]/10 hover:to-[#8B5CF6]/10 hover:border-[#E9D5FF] hover:text-[#8B5CF6]'
                }`}
              >
                {item}
                {isSelected && (
                  <span className="absolute inset-0 rounded-full bg-pink-400/10 blur-md animate-pulse pointer-events-none"></span>
                )}
              </span>
            )
          })}
        </div>

        <button
          type="submit"
          disabled={loading}
          className="w-full flex justify-center items-center gap-2 bg-gradient-to-r from-[#EC4899] to-[#8B5CF6] text-white px-4 py-2.5 mt-8 text-sm font-medium rounded-lg shadow-md transition-all duration-300 ease-in-out hover:opacity-90 active:scale-[0.98] disabled:opacity-60 disabled:cursor-not-allowed"
        >
          {loading ? (
            <span className="w-4 h-4 my-1 rounded-full border-2 border-t-transparent border-white animate-spin"></span>
          ) : (
            <Hash className="w-5" />
          )}
          Generate Title
        </button>
      </form>

      <div className="w-full max-w-lg p-6 bg-white rounded-xl border border-gray-200 shadow-sm hover:shadow-md transition-all duration-300 ease-out min-h-96 flex flex-col">
        <div className="flex items-center gap-3 mb-4">
          <Hash className="w-5 h-5 text-[#D946EF]" />
          <h1 className="text-xl font-semibold text-gray-800">
            Generated Titles
          </h1>
        </div>

        {!content ? (
          <div className="flex-1 flex justify-center items-center">
            <div className="text-sm flex flex-col items-center gap-5 text-gray-400 animate-fadeIn">
              <Hash className="w-10 h-10 text-[#D946EF] opacity-70" />
              <p className="text-center leading-relaxed">
                Enter a topic and click{' '}
                <span className="font-medium text-[#D946EF]">
                  "Generate Title"
                </span>{' '}
                to get started.
              </p>
            </div>
          </div>
        ) : (
          <div className="mt-3 h-full overflow-y-auto">
            <div className="prose prose-sm max-w-none text-slate-700">
              <Markdown>{content}</Markdown>
            </div>
          </div>
        )}
      </div>
    </div>
  )
}

export default BlogTitles

##Community.jsx
import { useUser } from '@clerk/clerk-react'
import React, { useEffect, useState } from 'react'
import { dummyPublishedCreationData } from '../assets/assets'
import { Heart } from 'lucide-react'

const Community = () => {
  const [creations, setCreations] = useState([])
  const { user } = useUser()

  const fetchCreations = async () => {
    setCreations(dummyPublishedCreationData)
  }

  useEffect(() => {
    if (user) {
      fetchCreations()
    }
  }, [user])

  const toggleLike = (index) => {
    setCreations((prev) =>
      prev.map((creation, i) => {
        if (i === index) {
          const hasLiked = creation.likes.includes(user?.id)
          return {
            ...creation,
            likes: hasLiked
              ? creation.likes.filter((id) => id !== user?.id)
              : [...creation.likes, user?.id],
          }
        }
        return creation
      })
    )
  }

  return (
    <div className='flex-1 h-full flex flex-col gap-4 p-6'>
      Creations
      <div className='bg-white h-full w-full rounded-xl overflow-y-scroll'>
        {creations.map((creation, index) => (
          <div
            key={index}
            className='relative group inline-block pl-3 pt-3 w-full sm:max-w-1/2 lg:max-w-1/3'
          >
            <img
              src={creation.content}
              alt=''
              className='w-full h-full object-cover rounded-lg'
            />

            <div
              className='absolute bottom-0 top-0 left-3 flex gap-2 items-end justify-end group-hover:justify-between p-3
              group-hover:bg-gradient-to-b from-transparent to-black/80 text-white rounded-lg'
            >
              <p className='text-sm hidden group-hover:block'>{creation.prompt}</p>
              <div className='flex gap-1 items-center'>
                <p>{creation.likes.length}</p>
                <Heart
                  onClick={() => toggleLike(index)}
                  className={`min-w-5 h-5 hover:scale-110 cursor-pointer ${
                    creation.likes.includes(user?.id)
                      ? 'fill-red-500 text-red-600'
                      : 'text-white'
                  }`}
                />
              </div>
            </div>
          </div>
        ))}
      </div>
    </div>
  )
}

export default Community

##GenerateImages.jsx
import { Image, Sparkles } from 'lucide-react'
import React, { useState } from 'react'
import axios from 'axios'
import { useAuth } from '@clerk/clerk-react'
import toast from 'react-hot-toast'

const api = axios.create({
  baseURL: 'http://localhost:3000',
  timeout: 120000,
  headers: {
    'Content-Type': 'application/json',
  },
})


const GenerateImages = () => {
  const blogCategories = [
    'Realistic',
    'Ghibli Style',
    'Anime Style',
    'Cartoon Style',
    'Fantasy Style',
    '3D Style',
    'Portrait Style',
  ]

  const [selectedStyle, setSelectedStyle] = useState('Realistic')
  const [input, setInput] = useState('')
  const [publish, setPublish] = useState(false)
  const [loading, setLoading] = useState(false)
  const [content, setContent] = useState('')

  const { getToken } = useAuth()

  const onSubmitHandler = async (e) => {
    e.preventDefault()

    if (!input.trim()) {
      toast.error('Please describe the image')
      return
    }

    try {
      setLoading(true)
      setContent('')

      const token = await getToken()

      if (!token) {
        toast.error('Authentication failed. Please refresh and try again.')
        return
      }

      const finalPrompt = `${input.trim()}, ${selectedStyle}`

      const { data } = await api.post(
        '/api/ai/generate-image',
        {
          prompt: finalPrompt,
          publish,
        },
        {
          headers: {
            Authorization: `Bearer ${token}`,
          },
        }
      )

      if (data?.success && data?.content) {
        setContent(data.content)
        toast.success('Image generated successfully')
      } else {
        toast.error(data?.message || 'Image generation failed')
      }
    } catch (error) {
      if (error.response) {
        if (error.response.status === 403) {
          toast.error('Premium feature only. Please upgrade your plan.')
        } else if (error.response.status === 400) {
          toast.error(error.response.data?.message || 'Invalid request')
        } else if (error.response.status === 401) {
          toast.error('Please login again')
        } else if (error.response.status === 429) {
          toast.error('Rate limit exceeded. Please try again later.')
        } else {
          toast.error(error.response.data?.message || 'Server error')
        }
      } else {
        toast.error('No response from server')
      }
    } finally {
      setLoading(false)
    }
  }

  return (
    <div className="h-full overflow-y-auto p-8 flex items-start flex-wrap gap-8 text-slate-700 bg-white">
      <form
        onSubmit={onSubmitHandler}
        className="w-full max-w-lg p-8 bg-white rounded-xl border border-gray-200 shadow-sm"
      >
        <div className="flex items-center gap-3 mb-4">
          <Sparkles className="w-6 h-6 text-[#00AD25]" />
          <h1 className="text-2xl font-semibold text-gray-800">
            AI Image Generator
          </h1>
        </div>

        <label className="block mt-6 text-sm font-medium text-gray-700">
          Describe Your Image
        </label>
        <textarea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          rows={4}
          className="w-full p-3 mt-2 text-sm rounded-lg border border-gray-300 outline-none focus:ring-2 focus:ring-[#00AD25]"
          placeholder="Describe what you want to see in the image"
          required
        />

        <p className="mt-6 text-sm font-medium text-gray-700">Style</p>
        <div className="mt-3 flex gap-3 flex-wrap">
          {blogCategories.map((item) => {
            const isSelected = selectedStyle === item
            return (
              <span
                key={item}
                onClick={() => setSelectedStyle(item)}
                className={`text-sm px-5 py-2 rounded-full border cursor-pointer transition ${
                  isSelected
                    ? 'bg-green-100 text-[#00AD25] border-[#00AD25]'
                    : 'border-gray-300 text-gray-700 hover:border-[#00AD25]'
                }`}
              >
                {item}
              </span>
            )
          })}
        </div>

        <div className="my-6 flex items-center gap-2">
          <label className="relative cursor-pointer">
            <input
              type="checkbox"
              checked={publish}
              onChange={(e) => setPublish(e.target.checked)}
              className="sr-only peer"
            />
            <div className="w-9 h-5 bg-slate-300 rounded-full peer-checked:bg-green-500 transition"></div>
            <span className="absolute left-1 top-1 w-3 h-3 bg-white rounded-full transition peer-checked:translate-x-4"></span>
          </label>
          <p className="text-sm">Make This Image Public</p>
        </div>

        <button
          type="submit"
          disabled={loading}
          className="w-full flex justify-center items-center gap-2 bg-gradient-to-r from-[#00AD25] to-[#04FF50] text-white px-4 py-2 mt-6 text-sm rounded-lg disabled:opacity-70"
        >
          {loading ? (
            <span className="w-4 h-4 my-1 rounded-full border-2 border-white border-t-transparent animate-spin"></span>
          ) : (
            <Image className="w-5" />
          )}
          Generate Image
        </button>
      </form>

      <div className="w-full max-w-lg p-6 bg-white rounded-xl border border-gray-200 shadow-sm min-h-96 flex flex-col">
        <div className="flex items-center gap-3 mb-4">
          <Image className="w-5 h-5 text-[#00AD25]" />
          <h1 className="text-xl font-semibold text-gray-800">
            Generated Image
          </h1>
        </div>

        {!content ? (
          <div className="flex-1 flex justify-center items-center">
            {loading ? (
              <span className="w-9 h-9 rounded-full border-4 border-[#00AD25] border-t-transparent animate-spin"></span>
            ) : (
              <div className="text-sm flex flex-col items-center gap-5 text-gray-400">
                <Image className="w-9 h-9" />
                <p className="text-center">
                  Enter a description and click Generate Image
                </p>
              </div>
            )}
          </div>
        ) : (
          <div className="mt-3 flex-1 flex flex-col">
            <img
              src={content}
              alt="Generated"
              className="w-full h-auto max-h-[400px] object-contain rounded-lg mb-4"
            />
            <div className="mt-4 p-3 bg-gray-50 rounded-lg">
              <p className="text-sm text-gray-600 mb-2">Image URL</p>
              <code className="text-xs text-gray-800 break-all">
                {content}
              </code>
            </div>
          </div>
        )}
      </div>
    </div>
  )
}

export default GenerateImages

  ]

    e.preventDefault()

    if (!isSignedIn) {
      toast.error('Please sign in to generate articles')
      return
    }

    if (loading) return
    if (input.trim().length < 3) {
      toast.error('Please enter a valid topic (minimum 3 characters)')
      return
    }

    try {
      setLoading(true)
      setContent('')

      console.log('Getting token...')
      const token = await getToken()
      console.log('Token received:', token ? 'Yes' : 'No')
      
      if (!token) {
        throw new Error('Authentication token not found. Please sign in again.')
      }

      console.log('Sending request to:', `${BASE_URL}/api/ai/generate-article`)
      
      const { data } = await axios.post(
        '/api/ai/generate-article',
        { 
          prompt: input, 
          length: selectedLength.length 
        },
        {
          baseURL: BASE_URL,
          headers: {
            Authorization: `Bearer ${token}`,
            'Content-Type': 'application/json',
          },
          timeout: 60000,
        }
      )

      console.log('Response received:', data)

      if (!data?.success) {
        throw new Error(data?.message || 'Article generation failed')
      }

      if (!data.content || data.content.trim().length === 0) {
        throw new Error('Received empty content from server')
      }

      setContent(data.content)
      toast.success('Article generated successfully!')
      
    } catch (error) {
      console.error('Error generating article:', {
        message: error.message,
        response: error.response?.data,
        status: error.response?.status
      })
      
      let errorMessage = 'Failed to generate article'
      
      if (error.response?.status === 401) {
        errorMessage = 'Authentication failed. Please sign in again.'
      } else if (error.response?.status === 402) {
        errorMessage = 'Free usage limit reached. Please upgrade to premium.'
      } else if (error.response?.data?.message) {
        errorMessage = error.response.data.message
      } else if (error.message) {
        errorMessage = error.message
      }
      
      toast.error(errorMessage)
    } finally {
      setLoading(false)
    }
  }

      const token = await getToken()

      console.error('Auth test failed:', error)
    }
  }

          ))}
        </div>

        <button
          type="submit"
          disabled={loading || !isSignedIn}
          className="w-full flex justify-center items-center gap-2 bg-gradient-to-r from-[#226BFF] to-[#65ADFF] text-white px-4 py-2 mt-6 text-sm rounded-lg disabled:opacity-70 disabled:cursor-not-allowed transition-opacity"
        >
          {loading ? (
            <>
              <span className="w-4 h-4 border-2 border-white border-t-transparent rounded-full animate-spin" />
              Generating...
            </>
          ) : !isSignedIn ? (
            'Sign in to Generate'
          ) : (
            <>
              <Edit className="w-5" />
              Generate article
            </>
          )}
        </button>
      </form>

      <div className="w-full max-w-lg p-4 bg-white rounded-lg flex flex-col border border-gray-200 min-h-96 max-h-[600px]">
        <div className="flex items-center gap-3">
          <Edit className="w-5 h-5 text-[#4A7AFF]" />
          <h1 className="text-xl font-semibold">Generated article</h1>
        </div>

        {loading ? (
          <div className="flex-1 flex flex-col justify-center items-center text-sm text-gray-400">
            <span className="w-6 h-6 border-2 border-blue-500 border-t-transparent rounded-full animate-spin mb-2" />
            Generating your article...
          </div>
        ) : !content ? (
          <div className="flex-1 flex justify-center items-center text-sm text-gray-400">
            {!isSignedIn ? 'Sign in to generate articles' : 'Enter a topic and generate an article'}
          </div>
        ) : (
          <div className="mt-3 h-full overflow-y-scroll text-sm text-slate-600 reset-tw">
            <Markdown>{content}</Markdown>
          </div>
        )}
      </div>
    </div>
  )
}

export default WriteArticle
